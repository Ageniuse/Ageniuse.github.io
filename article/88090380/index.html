这里是用企业开发标准还是用varchar comment '商品名称',(做备注)

这里id就不能使用int 自增类型 是有上限的 淘宝就不是这样id是用varchar的

create table 'product_info' (

--企业级的用varchar，自己玩的项目可以用自增的但数量大了可能不够用

‘product_id’ varchar(32) not null,(这里id就不能使用int 自增类型,是有上限的,淘宝就不是这样id是用varchar的)

'product_name' varchar(64) not null comment '商品名称',(做备注)

'product_price' decimal(8,2) not null comment '单价', (价格使用decimal)

‘product_stock’ int not null comment '库存',(需要加减)'

'product_description' varchar(64) comment '描述', (都是2的n次方)

'product_icon' varchar(512) comment '小图', （存url有点长）

'category_type' int not null comment '类目编号', （类目不会太多）

'create_time' timestamp not null default current_timestamp comment '创建时间',

(默认为当前时间 创建时间mysql自动写)

'update_time' timestamp not null default current_timestamp on update current_timestamp comment '修改时间',

--MYSQL5.7才可以default current_timestamp

(默认为当前时间 创建时间mysql自动写 on update current_timestamp 更新时自动更新)

（这里提醒一下 mysql5.6是不能2个字段都是 default current_timestamp，5.7就没错了 版本要一致和老师）

primary key ('product_id') (经常对id查询 建立主键 这样有索引)

)comment '商品表'；

create_time和update_time 加上 数据库处理 我们是不用处理的


类目不会太多 int 够了 自增

create table 'product_category'(

-- 类目不太可能爆多，所以可以自增

'category_name' varchar(64) not null comment '类目名字',

'category_type' int not null comment '类目编号',

'create_time' timestamp not null default current_timestamp comment '创建时间',

'update_time' timestamp not null default current_timestamp on update current_timestamp comment '修改时间',

-- MYSQL5.7才可以default current_timestamp

primary key ('category_id') ,

( 会经常根据类别查询 建立索引 不能有相同的类别 所以建立 约束索引)

unique key 'uqe_category_type' ('category_type') -- 类目是唯一的

) comment '类目表';
